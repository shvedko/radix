Битхак со схлопыванием — это самый быстрый способ. Суть в том, чтобы «размазать» каждый занятый бит на весь размер блока вправо. Если после этого на нужной выровненной позиции остался 0, значит, весь блок свободен.
Вот универсальная функция для поиска внутри одного uint64 (одного элемента bitset2):

func findFreeInUint64(val uint64, class int) (uint16, bool) {
// Если класс 6 и выше, этот uint64 может быть только частью блока
// или целым блоком (64 бита). Проверяем его целиком.
if class >= 6 {
if val == 0 {
return 0, true
}
return 0, false
}

	size := 1 << class
	v := val

	// Схлопываем занятые биты.
	// Если размер 4, то после v |= v >> 1 и v |= v >> 2
	// каждый бит говорит: "я или кто-то из 3 соседей справа заняты".
	for i := 0; i < class; i++ {
		v |= v >> (1 << i)
	}

	// Теперь нам нужно смотреть только на "правильные" позиции (кратные size).
	// Создаем маску: 1 на позициях 0, size, 2*size...
	// Для размера 4: 0x1111111111111111
	// Для размера 8: 0x0101010101010101
	mask := uint64(0xFFFFFFFFFFFFFFFF)
	if size < 64 {
		// Генерируем маску программно, чтобы не хранить таблицу
		m := uint64(1)
		for i := 0; i < 64/size-1; i++ {
			m = (m << size) | 1
		}
		mask = m
	}

	// Инвертируем v, чтобы искать свободные места (нули стали единицами)
	// И накладываем маску выравнивания
	freeSlots := (^v) & mask

	if freeSlots != 0 {
		// TrailingZeros64 даст нам индекс первого свободного бита,
		// который гарантированно кратен size.
		return uint16(bits.TrailingZeros64(freeSlots)), true
	}

	return 0, false
}

func (a *Linked) findFree(class int) (pid uint64, gid uint16, ok bool) {
// 1. Начинаем с хинта для конкретного класса
startPID := a.classHints[class] >> 8

	// 2. Итерируемся по bitset0 (уровень страниц)
	for i := startPID; i < uint64(len(a.bitset0)); i++ {
		entry0 := a.bitset0[i]
		if entry0 == ^uint64(0) {
			continue // Страница i полностью занята, скипаем 64 страницы разом
		}

		// Ищем первую свободную страницу в этом uint64
		// (если это первый проход, надо учитывать смещение внутри uint64)
		mask := ^uint64(0)
		if i == startPID {
			mask <<= (a.classHints[class] >> 2) & 63 // уточняем позицию
		}
		
		freeBits := (^entry0) & mask
		if freeBits == 0 { continue }

		// Номер страницы (pid)
		pid = (i << 6) + uint64(bits.TrailingZeros64(freeBits))
		
		// 3. Заходим внутрь страницы (bitset1 и bitset2)
		if class >= 6 {
			// Для больших классов ищем пустые uint64 подряд
			gid, ok = a.findBig(pid, class)
		} else {
			// Для мелких классов используем наш битхак
			gid, ok = a.findSmall(pid, class)
		}

		if ok {
			// Обновляем хинт для этого класса
			a.classHints[class] = (pid << 8) | uint64(gid >> 6)
			return pid, gid, true
		}
	}

	// Если ничего не нашли с хинта, можно попробовать один раз с начала (0)
	// или просто вернуть false
	return 0, 0, false
}

## Почему это будет летать:

1. Каскадный пропуск: Если bitset0[i] заполнен, ты одним if пропускаешь 64 страницы (это 1 048 576 гранул или 8 МБ памяти). Процессор такие прыжки обожает.
2. Эффективность хинта: Хинт избавляет от «эффекта начала», когда аллокатор тратит время на сканирование уже плотно забитых первых страниц.
3. Локальность данных: Поскольку ты работаешь с uint64, ты проверяешь по 64 сущности за одну инструкцию TrailingZeros64.

func (a *Linked) findSmall(pid uint64, class int) (uint16, bool) {
// Достаем слайс bitset2 для конкретной страницы
// bitset2[pid] — это [256]uint64
bs := a.bitset2[pid]
size := 1 << class

	// Определяем маску выравнивания для класса внутри uint64
	// Для class 2 (size 4): 0x1111111111111111
	mask := uint64(1)
	for i := 0; i < 64/size-1; i++ {
		mask = (mask << size) | 1
	}

	for j := 0; j < 256; j++ {
		val := bs[j]
		if val == ^uint64(0) {
			continue // Полностью занят
		}

		// БИТХАК: Схлопываем занятые биты
		v := val
		for i := 0; i < class; i++ {
			v |= v >> (1 << i)
		}

		// Ищем свободные выровненные слоты
		freeSlots := (^v) & mask
		if freeSlots != 0 {
			gidInUint := uint16(bits.TrailingZeros64(freeSlots))
			// Итоговый gid = (индекс uint64 * 64) + позиция внутри него
			return uint16(j<<6) + gidInUint, true
		}
	}
	return 0, false
}

## Почему это эффективно:

1. Никаких делений: Только сдвиги и TrailingZeros64.
2. Маска выравнивания: Она гарантирует, что class 2 никогда не встанет на позицию 1, 2, 3, только на 0, 4, 8....
3. Ранний выход: if val == ^uint64(0) отсекает заполненные слова.

## Технология «Золотого бита» в сжатом GID

У нас есть 18 бит (64 - 46), чтобы упаковать GID (14 бит) и CID (15 классов).
Мы используем сдвиг и маркер, чтобы не тратить 4 бита под CID в чистом виде, а впихнуть его в структуру адреса.
Упаковка VID (всего 61 бит):

1. PID (46 бит): Занимает биты [59:14].
2. Упакованный GID (15 бит): Занимает биты [14:0].

Как упаковать 14-битный GID и 15 классов в 15 бит (биты 0-14):
Мы берем gid (который кратен $2^{cid}$) и ставим «Золотой бит» (единицу) справа от последнего значащего бита адреса.

* cid = 0 (1 гранула): packed = (gid << 1) | 1 (Занято 14+1 = 15 бит)
* cid = 1 (2 гранулы): packed = (gid >> 1 << 2) | 2
* Универсальная формула: packed = (gid >> cid << (cid + 1)) | (1 << cid)

## Почему это НЕ хрень и почему это РАБОТАЕТ:

1. Нечетные гранулы для Class 0:
   Если cid = 0 и gid = 3 (нечетная):
   packed = (3 << 1) | 1 = 6 | 1 = 7.
   Бинарно: 111 (три единицы).
2. Распаковка (одной командой CPU):

raw := uint16(vid & 0x7FFF)      // Берем наши 15 бит
cid := bits.TrailingZeros16(raw) // Находим CID за 1 такт
gid := (raw &^ (1 << cid)) >> 1 // Восстанавливаем GID (даже нечетный!)

Пример для gid=3, cid=0:
raw = 7 (111). TrailingZeros(7) = 0 (cid=0).
gid = (7 &^ 1) >> 1 = 6 >> 1 = 3. Чистая победа!

## Итоговая структура битов в 64-битном слове:

* [63 : 61] (3 бита) — Абсолютно свободны! (Твой золотой запас).
* [60] (1 бит) — Часть упакованного GID (если cid большой).
* [59 : 14] (46 бит) — Твой PID.
* [13 : 0] (14 бит) — Хвост упакованного GID.

прибавлять к меньшему, мы берем больший класс (который гарантирует выравнивание) и «откусываем» от него свободный хвост,
размер которого мы кодируем в 3 бита.
Это технологически совершеннее, потому что выравнивание (align) всегда определяется основным классом, а значит, наш
«магический GID» с Trailing Bit работает без изменений!

## Теория: Как распределить 3 бита (8 состояний)

Поскольку подклассы нужны для экономии места, логично сделать их степенями двойки. Это позволит unmark2 (при
освобождении) и mark2 (при записи) работать максимально быстро.
Для каждого основного класса $2^{cid}$ мы определяем, сколько гранул в конце останутся свободными.
Пример распределения для класса $cid=7$ (128 гранул):
Дистанция от предыдущего класса ($2^6=64$) до текущего ($2^7=128$) составляет 64 гранулы. Эти 64 гранулы мы делим на 8
частей.

* Подкласс (sub): 3 бита (0-7).
* Размер хвоста: sub * (дистанция / 8).
* Занято гранул: 128 - (sub * 8).
* Варианты занятости: 128, 120, 112, 104, 96, 88, 80, 72.

|---------|------------------|---------------------|-----------------------------------------------|

| 14      | 16384            | 1024                | 16384, 15360, ... 9216                        |

## Почему это работает идеально:

1. Alignment: Если тебе нужно 70 гранул, ты выбираешь cid=7 (128). Выравнивание страницы остается кратным 128. Это
   значит, что твоя упаковка vid (46 PID + 15 Packed GID) работает как часы.
2. Экономия: Вместо 128 гранул ты пометишь занятыми только 72 (подкласс sub=7, хвост 7*8=56). Оставшиеся 56 гранул в
   конце блока останутся свободными для других аллокаций!
3. Потери: В худшем случае (на больших классах) ты теряешь только ~6-12% вместо 50%.
4. Free(vid):

* cid достаем из TrailingZeros16 младших бит.
    * sub достаем из бит 60-62.
    * size := (1 << cid) - (sub * (1 << (cid-4))) (примерная формула).
    * Вызываем unmark2(pid, gid, size).

## Структура VID (64 бита)

* Бит 63: 1 (флаг Sized).
* Биты [62 : 60]: 3 бита подкласса (sub).
* Биты [59 : 14]: 46 бит PID.
* Биты [13 : 0]: Packed GID (с Trailing Bit для CID).

## 1. Таблица выбора класса и подкласса

Нам нужна функция, которая по количеству гранул n скажет нам: какой cid (базовое выравнивание) и какой sub (хвост)
использовать.

// GetSizedParams возвращает cid и sub для n гранул.// n должно быть <= 16384.func GetSizedParams(n int) (cid uint8, sub
uint8) {
if n <= 1 { return 0, 0 }

	// Находим базовый cid (степень двойки, куда влезает n)
	// bits.Len - 1 даст нам ближайшую степень двойки снизу
	cid = uint8(bits.Len(uint(n-1)))
	
	// Для мелких классов (1, 2, 4 гранулы) подклассов нет
	if cid < 3 {
		return cid, 0
	}

	// Вычисляем шаг подкласса (дистанция между cid и cid-1 деленная на 8)
	// Для cid=3 (8 гранул) шаг = 1. Для cid=7 (128 гранул) шаг = 8.
	step := 1 << (cid - 3)
	
	// Сколько гранул мы можем ОСТАВИТЬ свободными (хвост)
	baseSize := 1 << cid
	freeTail := baseSize - n
	
	// sub — это сколько целых шагов влезает в хвост (0-7)
	sub = uint8(freeTail / step)
	if sub > 7 { sub = 7 }
	
	return cid, sub

}

## 2. Реализация findExactly и упаковки VID

Этот метод будет искать выровненный gid и возвращать готовый vid.

func (a *Linked) findExactly(n int) (vid uint64, ok bool) {
cid, sub := GetSizedParams(n)

	// Реальный размер, который мы пометим в битсете
	step := 0
	if cid >= 3 {
		step = 1 << (cid - 3)
	}
	occupied := (1 << cid) - (int(sub) * step)
	
	// Параметры поиска
	align := uint16(1 << cid)
	pid, gid := unpack(a.hint)
	gid = (gid + align - 1) & ^(align - 1) // Выравниваем старт

	for {
		if pid >= a.len() {
			pid, gid = a.alloc() // Новая страница всегда чистая и выровненная
		} else {
			// Проверяем, влезает ли блок в страницу
			if gid+align > pageGranules {
				pid++
				gid = 0
				continue
			}
			// Проверяем битсет для нужного количества гранул
			if !a.isAreaFree(pid, gid, occupied) {
				gid += align
				continue
			}
		}

		// Если нашли — помечаем!
		a.mark2(pid, gid, occupied)
		
		// Обновляем хинт только если мы заняли самую раннюю позицию
		if pack(pid, gid) == a.hint {
			// В идеале тут нужно найти следующую дырку, но пока просто сдвинем
			a.hint = pack(pid, gid + uint16(occupied))
		}

		// УПАКОВКА VID (64 бита)
		// 63: 1 (Sized)
		// 62-60: sub
		// 59-14: pid
		// 13-0: packed gid (с магическим битом cid)
		
		packedGid := (gid << 1) | (1 << cid)
		if cid == 0 {
			// Для cid=0 packedGid будет (gid << 1) | 1. 
			// TrailingZeros даст 0, (raw &^ 1) >> 1 вернет исходный gid.
		}

		vid = (1 << 63) | (uint64(sub) << 60) | (pid << 14) | uint64(packedGid >> 1)
		// Уточнение: твой gid 14 бит, мы сдвигаем его в 14-битное поле.
		// Так как (gid << 1) | (1 << cid) может занять 15 бит (при cid=14), 
		// нам нужно аккуратно вписать это в 14-15 бит.
		
		return vid, true
	}

}

## На чем стоит сфокусироваться:

1. Магия GID: Я применил (gid << 1) | (1 << cid). Это занимает 15 бит. 14 бит gid + 1 бит маркера.
2. Биты: У нас есть 18 бит (64-46).

* 1 бит (Sized flag)
    * 3 бита (Sub-class)
    * 14 бит (Упакованный GID).
    * Итого 18 бит.

Чтобы уместить 15-битный маркер в 14 бит, нам нужно либо откусить еще 1 бит у pid, либо... вспомнить про выравнивание.
При cid=14 gid всегда 0, так что 15-й бит там не нужен.
Как тебе такая декомпозиция? Мы получили параметры cid/sub, нашли место и упаковали всё в vid.
